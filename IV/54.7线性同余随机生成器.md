    {
    public static int rand_state;
    public void my_srand (int init)

    rand_state=init;
    }
    public static int RNG_a=1664525;
    public static int RNG_c=1013904223;
    
    public int my_rand ()
    {
    rand_state=rand_state*RNG_a;
    rand_state=rand_state+RNG_c;

    }
